    <script>
        // --- ESTRUCTURA DE DATOS (NO MODIFICAR VALORES) ---
        const RANGOS_PRECIOS = [
            // Multiplicadores de utilidad actualizados:
            { min: 1, max: 3, costo_maquina: 50.00, costo_perforacion: 20.00, multiplicador_utilidad: 6.0 },
            { min: 4, max: 10, costo_maquina: 20.00, costo_perforacion: 15.00, multiplicador_utilidad: 5.0 },
            { min: 11, max: 20, costo_maquina: 15.00, costo_perforacion: 12.00, multiplicador_utilidad: 4.0 },
            { min: 21, max: 50, costo_maquina: 14.00, costo_perforacion: 10.00, multiplicador_utilidad: 3.5 },
            { min: 51, max: 100, costo_maquina: 13.00, costo_perforacion: 8.00, multiplicador_utilidad: 3.0 },
            { min: 101, max: 200, costo_maquina: 12.00, costo_perforacion: 5.00, multiplicador_utilidad: 2.8 },
            { min: 201, max: 400, costo_maquina: 11.00, costo_perforacion: 3.00, multiplicador_utilidad: 2.5 },
            { min: 401, max: 800, costo_maquina: 10.00, costo_perforacion: 2.00, multiplicador_utilidad: 2.0 },
            { min: 801, max: 1000, costo_maquina: 9.00, costo_perforacion: 1.00, multiplicador_utilidad: 1.85 },
            { min: 1001, max: Infinity, costo_maquina: 8.00, costo_perforacion: 0.00, multiplicador_utilidad: 1.65 }
        ];

        const COSTOS_PRODUCCION = {
            tarjeta_virgen: 2.90,
            tarjeta_laminado: 4.50,
            limpieza: 1.00,
            impresion: {
                // $2,950 / 750 (1L) o 375 (2L)
                '1L': 2950 / 750, 
                '2L': 2950 / 375 
            },
            laminado_sencillo_u: 2250 / 750, // $2,250 / 750
            laminado_holografico_u: 3100 / 750 // $3,100 / 750
        };

        const COSTOS_DISENO = {
            'ClienteEnvia': 80.00,
            'Realizar': 250.00,
            'Guardado': 150.00
        };

        const COSTOS_BD_FIJOS = [
            { max: 100, costo: 250.00 },
            { max: 300, costo: 300.00 },
            { max: 500, costo: 350.00 },
            { max: 1000, costo: 400.00 },
            { max: Infinity, costo: 500.00 }
        ];

        const COSTO_BD_VARIABLE = 2.00;

        // Funci√≥n de formato para n√∫meros (ej: 1234.56 -> $1,234.56)
        const formatter = new Intl.NumberFormat('es-MX', {
            style: 'currency',
            currency: 'MXN',
            minimumFractionDigits: 2,
        });

        // --- FUNCI√ìN PARA TOGGLE (DESPLIEGUE) ---
        function toggleDesglose() {
            const content = document.getElementById('desgloseContent');
            const icon = document.getElementById('toggleIcon');
            const isHidden = content.classList.contains('hidden');
            
            if (isHidden) {
                content.classList.remove('hidden');
                icon.innerText = '‚ñ≤';
            } else {
                content.classList.add('hidden');
                icon.innerText = '‚ñº';
            }
        }

        // --- FUNCI√ìN PRINCIPAL DE C√ÅLCULO ---
        function calcularTotal() {
            const Q = parseInt(document.getElementById('cantidad').value) || 0;
            if (Q < 1) {
                document.getElementById('resultado').innerText = formatter.format(0);
                document.getElementById('desgloseContent').innerHTML = '<p class="text-center text-gray-500">Ingrese una cantidad v√°lida para ver el desglose.</p>';
                return;
            }

            // --- 1. Obtener Costos de Rango ---
            const rango = RANGOS_PRECIOS.find(r => Q >= r.min && Q <= r.max);
            
            const CUM = rango ? rango.costo_maquina : 8.00; 
            let CP = 0; // Costo Perforaci√≥n
            let costoPerforacionPorUnidad = 0;
            
            if (document.getElementById('perforacion').value === 'Si') {
                costoPerforacionPorUnidad = rango ? rango.costo_perforacion : 0.00;
                CP = costoPerforacionPorUnidad;
            }
            
            // --- 2. Costo Tarjeta Base (CTB) ---
            const laminadoOpcion = document.getElementById('laminadoOpcion').value;
            const CTB = (laminadoOpcion === 'Si') 
                ? COSTOS_PRODUCCION.tarjeta_laminado 
                : COSTOS_PRODUCCION.tarjeta_virgen;

            // Mostrar/Ocultar opciones de laminado
            const opcionesLamDiv = document.getElementById('opcionesLaminado');
            opcionesLamDiv.style.display = (laminadoOpcion === 'Si') ? 'block' : 'none';

            // --- 3. Costo Impresi√≥n (CI) y Laminado (CL) ---
            const tipoImpresion = document.getElementById('tipoImpresion').value;
            const CI = COSTOS_PRODUCCION.impresion[tipoImpresion];
            
            let CL = 0;
            let CL_FRENTE = 0;
            let CL_REVERSO = 0;
            
            if (laminadoOpcion === 'Si') {
                const lamFrente = document.getElementById('lamFrente').value;
                const lamReverso = document.getElementById('lamReverso').value;
                const advLaminado = document.getElementById('advLaminado');

                // Costo Laminado Frente
                if (lamFrente === 'Sencillo') {
                    CL_FRENTE = COSTOS_PRODUCCION.laminado_sencillo_u;
                } else if (lamFrente === 'Holografico') {
                    CL_FRENTE = COSTOS_PRODUCCION.laminado_holografico_u;
                }

                // Costo Laminado Reverso (solo sencillo)
                if (lamReverso === 'Sencillo') {
                    if (tipoImpresion === '2L') {
                        // Aplica costo de laminado
                        CL_REVERSO = COSTOS_PRODUCCION.laminado_sencillo_u; 
                        advLaminado.classList.add('hidden');
                    } else {
                        // No aplica costo de laminado en reverso si la impresi√≥n es 1L
                        advLaminado.classList.remove('hidden');
                    }
                } else {
                    advLaminado.classList.add('hidden');
                }
                
                CL = CL_FRENTE + CL_REVERSO;
            }

            // --- 4. TBP y P.V.U. ---
            const CLIM = COSTOS_PRODUCCION.limpieza;
            
            // TBP_SinUtilidad: Suma de costos de producci√≥n directos (Tarjeta, Limpieza, Impresi√≥n, Laminado)
            const TBP_SinUtilidad = CTB + CLIM + CI + CL; 
            
            // Obtener Multiplicador de Utilidad (MU)
            const multiplicadorUtilidad = rango ? rango.multiplicador_utilidad : 1.65; // Usar el valor m√≠nimo si Q es muy alto

            // TBP_Utilidad: Costo base de producci√≥n con el margen de utilidad aplicado
            const TBP_Utilidad = TBP_SinUtilidad * multiplicadorUtilidad;
            const UtilidadPorUnidad = TBP_Utilidad - TBP_SinUtilidad;

            // PVU: Precio de Venta Unitario (TBP con utilidad + Costo M√°quina + Costo Perforaci√≥n)
            const PVU = TBP_Utilidad + CUM + CP; 

            // --- 5. TCU (Total Costo Unitario) ---
            let TCU = Q * PVU;

            // --- 6. Costo de Dise√±o (CD) ---
            let CD = 0;
            const advDiseno = document.getElementById('advDiseno');
            const disenoOpcion = document.getElementById('opcionDiseno').value;
            let disenoExplicacion = document.getElementById('opcionDiseno').options[document.getElementById('opcionDiseno').selectedIndex].text;
            
            if (Q > 100) {
                CD = 0;
                advDiseno.classList.remove('hidden');
                disenoExplicacion = 'Gratis por superar 100 tarjetas.';
            } else {
                CD = COSTOS_DISENO[disenoOpcion] || 0;
                advDiseno.classList.add('hidden');
            }

            // --- 7. Costo Base de Datos (CBD) ---
            let CBD = 0;
            let fijoBDCosto = 0;
            let variablesCosto = 0;
            let numVariables = 0;
            const llevaBD = document.getElementById('llevaBD').value;
            const opcionesBDDiv = document.getElementById('opcionesBD');
            const fijoBDValueSpan = document.getElementById('fijoBDValue');
            opcionesBDDiv.style.display = (llevaBD === 'Si') ? 'block' : 'none';

            if (llevaBD === 'Si') {
                // Costo Fijo BD
                const fijoBD = COSTOS_BD_FIJOS.find(r => Q <= r.max);
                fijoBDCosto = fijoBD ? fijoBD.costo : COSTOS_BD_FIJOS[COSTOS_BD_FIJOS.length - 1].costo;

                fijoBDValueSpan.innerText = formatter.format(fijoBDCosto);
                CBD += fijoBDCosto;

                // Costo Variable BD
                numVariables = parseInt(document.getElementById('numVariables').value) || 0;
                variablesCosto = numVariables * COSTO_BD_VARIABLE;
                CBD += variablesCosto;
            } else {
                 fijoBDValueSpan.innerText = formatter.format(0);
            }

            // --- 8. Total Final ---
            const totalFinal = TCU + CD + CBD;
            document.getElementById('resultado').innerText = formatter.format(totalFinal);
            
            // --- 9. Generar Contenido del Desglose ---
            let htmlDesglose = `
                <div class="space-y-4">
                    <h4 class="font-bold text-lg text-indigo-700">1. COSTO POR UNIDAD (P.V.U. = ${formatter.format(PVU)})</h4>
                    
                    <div class="ml-4 p-2 bg-white rounded-md border border-gray-200">
                        <p class="font-semibold text-base mb-1">Costo Base de Producci√≥n (TBP sin utilidad): ${formatter.format(TBP_SinUtilidad)}</p>
                        <ul class="list-disc ml-6 text-xs space-y-1">
                            <li>Tarjeta Virgen: ${formatter.format(CTB)} (${laminadoOpcion === 'Si' ? 'Preparada para Laminar' : 'Est√°ndar'})</li>
                            <li>Costo Limpieza/Manejo: ${formatter.format(CLIM)}</li>
                            <li>Costo Cinta de Impresi√≥n (${tipoImpresion}): ${formatter.format(CI)}</li>
                            <li>Costo Laminado (Frente): ${formatter.format(CL_FRENTE)}</li>
                            ${CL_REVERSO > 0 ? `<li>Costo Laminado (Reverso): ${formatter.format(CL_REVERSO)}</li>` : ''}
                        </ul>
                    </div>

                    <div class="ml-4 p-2 bg-yellow-50 rounded-md border border-yellow-300">
                        <p class="font-semibold text-base mb-1 text-yellow-800">C√°lculo de Utilidad / Margen por Unidad:</p>
                        <ul class="list-disc ml-6 text-xs space-y-1 text-yellow-700">
                            <li>Multiplicador por Volumen (Rango ${rango.min}-${rango.max}): <span class="font-bold text-base">x${multiplicadorUtilidad.toFixed(2)}</span></li>
                            <li>Costo Base de Producci√≥n con Utilidad (TBP Utilidad): ${formatter.format(TBP_Utilidad)}</li>
                            <li><span class="font-bold">Utilidad por Unidad: ${formatter.format(UtilidadPorUnidad)}</span> (TBP Utilidad - TBP sin Utilidad)</li>
                        </ul>
                    </div>

                    <div class="ml-4 p-2 bg-white rounded-md border border-gray-200">
                        <p class="font-semibold text-base mb-1">Costos de Operaci√≥n por Unidad (Sin Multiplicador):</p>
                        <ul class="list-disc ml-6 text-xs space-y-1">
                            <li>Costo Uso M√°quina (CUM): ${formatter.format(CUM)} (Aplicado al rango ${rango.min}-${rango.max})</li>
                            <li>Costo Perforaci√≥n (CP): ${formatter.format(costoPerforacionPorUnidad)} (Aplicado si se seleccion√≥ perforaci√≥n)</li>
                        </ul>
                    </div>

                    <p class="font-bold text-lg text-indigo-700 pt-3">2. COSTO TOTAL UNITARIO (TCU): ${formatter.format(TCU)}</p>
                    <p class="text-sm text-gray-600 ml-4">= ${formatter.format(Q)} tarjetas x ${formatter.format(PVU)} P.V.U.</p>

                    <h4 class="font-bold text-lg text-red-700 pt-3">3. CARGOS FIJOS DEL PEDIDO: ${formatter.format(CD + CBD)}</h4>
                    
                    <div class="ml-4 p-2 bg-white rounded-md border border-gray-200">
                        <p class="font-semibold text-base mb-1">Costo de Dise√±o: ${formatter.format(CD)}</p>
                        <ul class="list-disc ml-6 text-xs space-y-1">
                            <li>Opci√≥n: ${disenoExplicacion}</li>
                        </ul>
                    </div>
                    
                    <div class="ml-4 p-2 bg-white rounded-md border border-gray-200">
                        <p class="font-semibold text-base mb-1">Costo Base de Datos (Personalizaci√≥n): ${formatter.format(CBD)}</p>
                        <ul class="list-disc ml-6 text-xs space-y-1">
                            <li>Costo Fijo por Rango (${Q} unidades): ${formatter.format(fijoBDCosto)}</li>
                            <li>Costo Variable (Variables Distintas): ${formatter.format(variablesCosto)} (${numVariables} variables x $2.00)</li>
                        </ul>
                    </div>
                </div>
            `;
            
            document.getElementById('desgloseContent').innerHTML = htmlDesglose;
        }

        // Ejecutar el c√°lculo al cargar la p√°gina para mostrar un valor inicial
        document.addEventListener('DOMContentLoaded', calcularTotal);
