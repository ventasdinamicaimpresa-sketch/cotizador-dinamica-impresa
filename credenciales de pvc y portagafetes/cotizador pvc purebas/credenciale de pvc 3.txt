        });

        // Funci√≥n de formato num√©rico simple (ej: 1.2345 -> 1.23)
        const numFormatter = new Intl.NumberFormat('es-MX', {
            minimumFractionDigits: 2,
            maximumFractionDigits: 4,

        function copyCotizacion() {
            const quoteText = document.getElementById('whatsappQuoteText');
            // Usamos un peque√±o truco para asegurar que solo se copia el contenido sin formato extra
            const tempInput = document.createElement('textarea');
            tempInput.value = quoteText.value;
            document.body.appendChild(tempInput);
            tempInput.select();
            tempInput.setSelectionRange(0, 99999);
            
            let success = false;
            try {
                success = document.execCommand('copy');
            } catch (err) {
                console.error('No se pudo copiar el texto.', err);
            } finally {
                document.body.removeChild(tempInput);
            }

            const button = document.getElementById('copyButton');
            button.classList.remove('bg-green-600', 'bg-red-600', 'bg-blue-600');
            
            if (success) {
                button.innerText = '‚úÖ ¬°Copiado!';
                button.classList.add('bg-green-600');
            } else {
                button.innerText = '‚ùå Error al Copiar';
                button.classList.add('bg-red-600');
            }
            
            setTimeout(() => {
                button.innerText = 'üìã Copiar Cotizaci√≥n';
                button.classList.remove('bg-green-600', 'bg-red-600');
                button.classList.add('bg-blue-600');
            }, 2000);
        }

        // --- FUNCI√ìN PRINCIPAL DE C√ÅLCULO ---
        function calcularTotal() {
            const Q = parseInt(document.getElementById('cantidad').value) || 0;
            const quoteTextElement = document.getElementById('whatsappQuoteText');
            const copyButton = document.getElementById('copyButton');

            // Resetear bot√≥n de copiar
            copyButton.innerText = 'üìã Copiar Cotizaci√≥n';
            copyButton.classList.remove('bg-green-600', 'bg-red-600');
            copyButton.classList.add('bg-blue-600');


            if (Q < 1) {
                // Aqu√≠ podr√≠as agregar l√≥gica para mostrar un mensaje de error si Q es inv√°lida
                quoteTextElement.value = "Error: Cantidad inv√°lida.";
                return;
            }

            // --- 1. Obtener Costos de Rango (CUM, CP, Multiplicador) ---
            const rango = RANGOS_PRECIOS.find(r => Q >= r.min && Q <= r.max);
            
            const CUM = rango ? rango.costo_maquina : 8.00; 
            let CP = 0; 
            
            if (document.getElementById('perforacion').value === 'Si') {
                CP = rango ? rango.costo_perforacion : 0.00; 
            }
            const multiplicadorUtilidad = rango ? rango.multiplicador_utilidad : 1.65; 

            // --- 2. Costo Tarjeta Base (CTB) ---
            const laminadoOpcion = document.getElementById('laminadoOpcion').value;
            const CTB_Inicial = (laminadoOpcion === 'Si') 
                ? COSTOS_PRODUCCION.tarjeta_laminado 
                : COSTOS_PRODUCCION.tarjeta_virgen;
            
            // Mostrar/Ocultar opciones de laminado y actualizar rangos de f√≥rmulas
            const opcionesLamDiv = document.getElementById('opcionesLaminado');
            opcionesLamDiv.style.display = (laminadoOpcion === 'Si') ? 'block' : 'none';
            
            document.getElementById('m_min').innerText = RANGOS_PRECIOS[RANGOS_PRECIOS.length - 1].multiplicador_utilidad.toFixed(2);
            document.getElementById('m_max').innerText = RANGOS_PRECIOS[0].multiplicador_utilidad.toFixed(2);
            document.getElementById('c_min').innerText = formatter.format(RANGOS_PRECIOS[RANGOS_PRECIOS.length - 1].costo_maquina);
            document.getElementById('c_max').innerText = formatter.format(RANGOS_PRECIOS[0].costo_maquina);


            // --- 3. Costo Impresi√≥n (CI) y Limpieza (CLIM) ---
            const tipoImpresion = document.querySelector('input[name="tipoImpresionRadios"]:checked').value;
            const CI = COSTOS_PRODUCCION.impresion[tipoImpresion];
            const CLIM = COSTOS_PRODUCCION.limpieza;
            
            // --- 4. Costo Laminado (CL) ---
            let CL = 0; 
            let Markup50 = 0;
            let acabadoText = 'sin acabados';

            if (laminadoOpcion === 'Si') {
                const lamFrente = document.getElementById('lamFrente').value;
                const lamReverso = document.getElementById('lamReverso').value;
                const advLaminado = document.getElementById('advLaminado');

                let CL_FRENTE = 0;
                let CL_REVERSO = 0;

                if (lamFrente === 'Sencillo') {
                    CL_FRENTE = COSTOS_PRODUCCION.laminado_sencillo_u;
                    acabadoText = 'con laminado sencillo (frente)';
                } else if (lamFrente === 'Holografico') {
                    CL_FRENTE = COSTOS_PRODUCCION.laminado_holografico_u;
                    acabadoText = 'con hologr√°fico (frente)';
                }

                if (lamReverso === 'Sencillo') {
                    if (tipoImpresion === '2L') {
                        CL_REVERSO = COSTOS_PRODUCCION.laminado_sencillo_u; 
                        advLaminado.classList.add('hidden');
                        acabadoText = acabadoText.includes('con ') ? acabadoText + ' y laminado sencillo (reverso)' : 'con laminado sencillo (frente y reverso)';
                    } else {
                        advLaminado.classList.remove('hidden');
                        // Se ignora el costo de reverso si es 1L, pero se mantiene el texto si aplica al frente.
                    }
                } else {
                    advLaminado.classList.add('hidden');
                }
                
                CL = CL_FRENTE + CL_REVERSO;

                // Aplicaci√≥n de 50% Markup a Materiales (CTB_Inicial + CL)
                const CostoMaterialLaminado = CTB_Inicial + CL; 
                Markup50 = CostoMaterialLaminado * 0.50; 
                
                document.getElementById('d_markupRow').style.display = 'flex';
                document.getElementById('d_markupValue').innerText = formatter.format(Markup50);

            } else {
                document.getElementById('d_markupRow').style.display = 'none';
                document.getElementById('d_markupValue').innerText = formatter.format(0);
            }

            if (document.getElementById('perforacion').value === 'Si') {
                 acabadoText = acabadoText.includes('con ') ? acabadoText + ' y perforaci√≥n' : 'con perforaci√≥n';
            }
            if (acabadoText === 'sin acabados') {
                acabadoText = 'sin acabados';
            }


            // --- 5. TBP (Total Base Producci√≥n) y PVU ---
            // TBP = (CTB + CL + Markup50) + CLIM + CI
            const TBP_SinUtilidad = CTB_Inicial + CL + Markup50 + CLIM + CI;
            
            // PVU_Base = TBP * Multiplicador
            const PVU_Base = TBP_SinUtilidad * multiplicadorUtilidad;
            
            // PVU Final = PVU_Base + CUM + CP
            const PVU = PVU_Base + CUM + CP; 

            // PVU con IVA
            const PVU_CON_IVA = PVU * (1 + IVA_RATE);

            // --- 6. TCU (Total Costo Unitario) ---
            const TCU = Q * PVU;

            // --- 7. Costos Fijos (Dise√±o y BD) (CD, CBD) ---
            let CD = 0;
            const advDiseno = document.getElementById('advDiseno');
            const disenoOpcion = document.getElementById('opcionDiseno').value;
            let disenoTexto = '';
            
            if (Q > 100) { 
                CD = 0; 
                advDiseno.classList.remove('hidden');
                disenoTexto = 'Dise√±o Gratis (Por volumen)';
            } else { 
                CD = COSTOS_DISENO[disenoOpcion] || 0; 
                advDiseno.classList.add('hidden'); 
                
                if (disenoOpcion === 'ClienteEnvia') disenoTexto = 'cliente envia dise√±o';
                else if (disenoOpcion === 'Realizar') disenoTexto = 'dise√±o realizado por nosotros';
                else if (disenoOpcion === 'Guardado') disenoTexto = 'dise√±o guardado aqu√≠';
            }

            let CBD = 0;
            let fijoBDCosto = 0;
            let numVariables = 0;
            const llevaBD = document.getElementById('llevaBD').value;
            const opcionesBDDiv = document.getElementById('opcionesBD');
            const fijoBDValueSpan = document.getElementById('fijoBDValue');
            opcionesBDDiv.style.display = (llevaBD === 'Si') ? 'block' : 'none';
            let bdText = '';


            if (llevaBD === 'Si') {
                const fijoBD = COSTOS_BD_FIJOS.find(r => Q <= r.max);
                fijoBDCosto = fijoBD ? fijoBD.costo : COSTOS_BD_FIJOS[COSTOS_BD_FIJOS.length - 1].costo;
                numVariables = parseInt(document.getElementById('numVariables').value) || 0;
                
                CBD = fijoBDCosto + (numVariables * COSTO_BD_VARIABLE);
                fijoBDValueSpan.innerText = formatter.format(fijoBDCosto);
                bdText = ` + Personalizaci√≥n BD (${numVariables} variables)`;
            } else {
                 fijoBDValueSpan.innerText = formatter.format(0);
            }

            // --- 8. Totales finales ---
            const subtotalSinIva = TCU + CD + CBD; 
            const totalConIva = subtotalSinIva * (1 + IVA_RATE);

            // --- 9. Actualizar Desglose de Resultados (Visual) ---
            document.getElementById('pvuSinIva').innerText = formatter.format(PVU);
            document.getElementById('pvuConIva').innerText = formatter.format(PVU_CON_IVA); 
            document.getElementById('subtotalSinIva').innerText = formatter.format(subtotalSinIva);
            document.getElementById('totalConIva').innerText = formatter.format(totalConIva);
            
            // --- 10. Actualizar el Desglose Detallado (Mejorado) ---
            document.getElementById('d_costoBase').innerText = formatter.format(CTB_Inicial);
            document.getElementById('d_costoBaseM').innerText = CTB_Inicial.toFixed(2); 
            document.getElementById('d_costoLaminado').innerText = formatter.format(CL);
            document.getElementById('d_costoLamTipo').innerText = CL.toFixed(2);
            document.getElementById('d_costoLimpieza').innerText = formatter.format(CLIM);
            document.getElementById('d_costoImpresion').innerText = formatter.format(CI);
            
            document.getElementById('d_subtotalCosto').innerText = formatter.format(TBP_SinUtilidad);
            document.getElementById('d_multiplicador').innerText = numFormatter.format(multiplicadorUtilidad);
            document.getElementById('d_utilidadApplied').innerText = formatter.format(PVU_Base - TBP_SinUtilidad); // Muestra la utilidad en s√≠ (PVU_Base - TBP)
            document.getElementById('d_costoMaquina').innerText = formatter.format(CUM);
            document.getElementById('d_costoPerforacion').innerText = formatter.format(CP);
            document.getElementById('d_pvuFinal').innerText = formatter.format(PVU);


            // --- 11. Generar Contenido de la Cotizaci√≥n (Formato Exacto Solicitado) ---
            
            // PVU: 2 decimales, SIN separador de miles.
            const pvuString = PVU_CON_IVA.toFixed(2); 
            
            // TOTAL: 2 decimales, CON separador de miles.
            const totalStringFormatted = totalFormatter.format(totalConIva);
            
            const ladosTexto = tipoImpresion === '1L' ? '1 lado' : '2 lados';

            // ** FORMATO SOLICITADO POR EL USUARIO EN UNA SOLA L√çNEA (con separador de miles en total y = neto) **
            const resumenWhatsapp = 
                `${Q} tarjeta de pvc medida 8.56x5.4 cm grosor de 30 mil. impresas a color ${ladosTexto} ${acabadoText}${bdText}, ${disenoTexto} = $${pvuString} c/u , total = $${totalStringFormatted} = neto`;

            // Actualizar el √°rea de texto con la cotizaci√≥n
            quoteTextElement.value = resumenWhatsapp;
        }

        // Ejecutar el c√°lculo al cargar la p√°gina para mostrar un valor inicial
        document.addEventListener('DOMContentLoaded', calcularTotal);
